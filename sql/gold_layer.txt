-- SALES OVERVIEW
CREATE OR REPLACE TABLE SUPPLY.GOLD.SALES AS
SELECT 
    YEAR, MONTH, MONTH_NAME, QUARTER,CUSTOMER_NAME, CUSTOMER_SEGMENT, CUSTOMER_INDUSTRY, CUSTOMER_CITY, PR_BRAND, PR_CATEGORY, PR_SUBCATEGORY, S_COUNTRY, WH_PROVINCE, FULL_DATE,
    
    COUNT(DISTINCT ORDER_ID) TOTAL_ORDERS,
    SUM(QUANTITY_ORDERED) TOTAL_UNITS_SOLD,
    SUM(GROSS_AMOUNT) TOTAL_GROSS_AMOUNT,
    SUM(REVENUE) TOTAL_REVENUE,
    SUM(NET_REVENUE) TOTAL_NET_REVENUE,
    SUM(TOTAL_COST) TOTAL_COST,
    SUM(GROSS_PROFIT) TOTAL_GROSS_PROFIT,
    ROUND(AVG(GROSS_PROFIT_MARGIN_PCT), 2) AVG_GROSS_PROFIT_MARGIN,
    ROUND(AVG(DISCOUNT_PCT), 2) AVG_DISCOUNT_PCT,
    SUM(SHIPPING_COST) TOTAL_SHIPPING_COST,
    
    CURRENT_TIMESTAMP::TIMESTAMP_NTZ INGESTION_TIMESTAMP, 'SILVER.ORDERS' SOURCE_SYSTEM
FROM SUPPLY.SILVER.ORDERS
GROUP BY YEAR, MONTH, MONTH_NAME, QUARTER, CUSTOMER_NAME, CUSTOMER_SEGMENT, CUSTOMER_INDUSTRY, CUSTOMER_CITY, PR_BRAND, PR_CATEGORY, PR_SUBCATEGORY, S_COUNTRY, WH_PROVINCE, FULL_DATE
ORDER BY YEAR, MONTH;

-- SALES BY CATEGORY
CREATE OR REPLACE TABLE SUPPLY.GOLD.SALES_BY_CATEGORY AS
SELECT 
    PR_CATEGORY, PR_SUBCATEGORY,FULL_DATE,

    COUNT(DISTINCT ORDER_ID) TOTAL_ORDERS,
    COUNT(DISTINCT PRODUCT_KEY) TOTAL_PRODUCTS,
    SUM(QUANTITY_ORDERED) TOTAL_UNIT_SOLD,
    SUM(REVENUE) TOTAL_REVENUE,
    SUM(GROSS_PROFIT) TOTAL_GROSS_PROFIT,
    ROUND(AVG(GROSS_PROFIT_MARGIN_PCT), 2) AVG_GROSS_PROFIT_MARGIN_PCT,
    ROUND(AVG(DISCOUNT_PCT), 2) AVG_DISCOUNT_PCT,
    CURRENT_TIMESTAMP::TIMESTAMP_NTZ INGESTION_TIMESTAMP, 'SILVER.ORDERS' SOURCE_SYSTEM
FROM SUPPLY.SILVER.ORDERS
GROUP BY PR_CATEGORY, PR_SUBCATEGORY, FULL_DATE
ORDER BY TOTAL_REVENUE DESC;

-- CUSTOMER
CREATE OR REPLACE TABLE SUPPLY.GOLD.CUSTOMER AS
SELECT
    CUSTOMER_KEY, CUSTOMER_NAME, CUSTOMER_SEGMENT, CUSTOMER_INDUSTRY, CUSTOMER_CITY,FULL_DATE,

    COUNT(DISTINCT ORDER_ID) TOTAL_ORDERS,
    SUM(QUANTITY_ORDERED) TOTAL_UNIT_PURCHASED,
    SUM(REVENUE) TOTAL_REVENUE,
    ROUND(AVG(REVENUE), 2) AVG_ORDER_VALUE,
    SUM(GROSS_PROFIT) TOTAL_GROSS_PROFIT,
    ROUND(AVG(DISCOUNT_PCT), 2) AVG_DISCOUNT_PCT,
    CURRENT_TIMESTAMP::TIMESTAMP_NTZ INGESTION_TIMESTAMP, 'SILVER.ORDERS' SOURCE_SYSTEM
FROM SUPPLY.SILVER.ORDERS
GROUP BY CUSTOMER_KEY, CUSTOMER_NAME, CUSTOMER_SEGMENT, CUSTOMER_INDUSTRY, CUSTOMER_CITY, FULL_DATE
ORDER BY TOTAL_REVENUE DESC;

-- SUPPLIER
CREATE OR REPLACE TABLE SUPPLY.GOLD.SUPPLIER AS
SELECT
    SUPPLIER_KEY, SUPPLIER_NAME, CONTRACT_TYPE, SUPPLIER_RATING, LEAD_TIME_DAYS,FULL_DATE,

    COUNT(DISTINCT ORDER_ID) TOTAL_ORDERS,
    SUM(QUANTITY_ORDERED) TOTAL_UNIT_SUPPLIED,
    SUM(REVENUE) TOTAL_REVENUE,
    CURRENT_TIMESTAMP::TIMESTAMP_NTZ INGESTION_TIMESTAMP, 'SILVER.ORDERS' SOURCE_SYSTEM
FROM SUPPLY.SILVER.ORDERS
GROUP BY SUPPLIER_KEY, SUPPLIER_NAME, CONTRACT_TYPE, SUPPLIER_RATING, LEAD_TIME_DAYS, FULL_DATE
ORDER BY TOTAL_REVENUE;

-- WAREHOUSE
CREATE OR REPLACE TABLE SUPPLY.GOLD.WAREHOUSE AS
SELECT 
    WAREHOUSE_KEY, WAREHOUSE_NAME, WH_PROVINCE WAREHOUSE_PROVINCE, WH_CITY WAREHOUSE_CITY, WH_MANAGER_NAME WAREHOUSE_MANAGER_NAME, FULL_DATE,
    COUNT(DISTINCT ORDER_ID) TOTAL_ORDERS,
    SUM(QUANTITY_ORDERED) TOTAL_UNIT_PROCESSED,
    SUM(REVENUE) TOTAL_REVENUE,
    ROUND(AVG(LEAD_TIME_DAYS), 0) AVG_LEAD_TIME_DAYS,
    CURRENT_TIMESTAMP::TIMESTAMP_NTZ INGESTION_TIMESTAMP, 'SILVER.ORDERS' SOURCE_SYSTEM
FROM SUPPLY.SILVER.ORDERS
GROUP BY WAREHOUSE_KEY, WAREHOUSE_NAME, WH_PROVINCE, WH_CITY , WH_MANAGER_NAME, FULL_DATE
ORDER BY TOTAL_REVENUE DESC;

-- INVENTORY
CREATE OR REPLACE TABLE SUPPLY.GOLD.INVENTORY AS
SELECT
    WAREHOUSE_KEY,
    WAREHOUSE_NAME, STOCK_ON_HAND, CAPACITY,
    WH_PROVINCE AS WAREHOUSE_PROVINCE,
    WH_CITY AS WAREHOUSE_CITY,
    YEAR,
    QUARTER,
    MIN(FULL_DATE) AS FULL_DATE,
    SUM(STOCK_ON_HAND) AS STOCK_ON_HAND,
    SUM(REORDER_LEVEL) AS REORDER_LEVEL,
    SUM(UNITS_ON_ORDER) AS UNITS_ON_ORDER,
    SUM(STOCK_VALUE) AS STOCK_VALUE,
    SUM(STOCK_BUFFER) AS STOCK_BUFFER,
    ROUND(AVG(STOCK_VS_REORDER_PCT), 2) AS STOCK_VS_REORDER_PCT,
    ROUND(SUM(STOCK_ON_HAND) / NULLIF(MAX(CAPACITY), 0) * 100, 2) AS WAREHOUSE_UTILIZATION_PCT,
    MAX(CAPACITY) AS WAREHOUSE_CAPACITY,
    CURRENT_TIMESTAMP::TIMESTAMP_NTZ AS INGESTION_TIMESTAMP,
    'SILVER.INVENTORY' AS SOURCE_SYSTEM
FROM SUPPLY.SILVER.INVENTORY
GROUP BY
    WAREHOUSE_KEY, WAREHOUSE_NAME, WH_PROVINCE, WH_CITY, YEAR, QUARTER, STOCK_ON_HAND, CAPACITY
ORDER BY YEAR, QUARTER, WAREHOUSE_KEY;

-- DIM DATE
CREATE OR REPLACE TABLE SUPPLY.GOLD.DIM_DATE AS
SELECT
    DATE_KEY,
    FULL_DATE,
    DAY,
    MONTH,
    MONTH_NAME,
    QUARTER,
    YEAR,
    WEEK_NUMBER,
    IS_WEEKEND
FROM SUPPLY.BRONZE.DATE
ORDER BY FULL_DATE;
