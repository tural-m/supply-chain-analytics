--ORDERS
CREATE OR REPLACE TABLE SUPPLY.SILVER.ORDERS AS
WITH CLEAN_CUSTOMER AS (
    SELECT*
    FROM(
        SELECT *
            REPLACE(
                UPPER(TRIM(CUSTOMER_NAME)) AS CUSTOMER_NAME,
                UPPER(TRIM(SEGMENT)) AS SEGMENT,
                UPPER(TRIM(INDUSTRY)) AS INDUSTRY,
                UPPER(TRIM(COUNTRY)) AS COUNTRY,
                UPPER(TRIM(CITY)) AS CITY
            ),
            ROW_NUMBER() OVER(PARTITION BY CUSTOMER_KEY ORDER BY CUSTOMER_KEY) AS RN
        FROM SUPPLY.BRONZE.CUSTOMER
        WHERE CUSTOMER_KEY IS NOT NULL
        ) as SQ
    WHERE RN = 1
),
CLEAN_PRODUCT AS (
    SELECT *
    FROM (
        SELECT *
            REPLACE(
                UPPER(TRIM(PRODUCT_NAME)) AS PRODUCT_NAME,
                UPPER(TRIM(CATEGORY)) AS CATEGORY,
                UPPER(TRIM(SUBCATEGORY)) AS SUBCATEGORY,
                UPPER(TRIM(BRAND)) AS BRAND
            ),
            ROW_NUMBER() OVER(PARTITION BY PRODUCT_KEY ORDER BY PRODUCT_KEY) AS RN
        FROM SUPPLY.BRONZE.PRODUCT
        WHERE PRODUCT_KEY IS NOT NULL
        ) AS SQ
    WHERE RN = 1
),
CLEAN_SUPPLIER AS (
    SELECT *
    FROM (
        SELECT *
            REPLACE(
                UPPER(TRIM(SUPPLIER_NAME)) AS SUPPLIER_NAME,
                UPPER(TRIM(COUNTRY)) AS COUNTRY,
                UPPER(TRIM(CONTRACT_TYPE)) AS CONTRACT_TYPE
            ),
            ROW_NUMBER() OVER(PARTITION BY SUPPLIER_KEY ORDER BY SUPPLIER_KEY) AS RN
        FROM SUPPLY.BRONZE.SUPPLIER
        WHERE SUPPLIER_KEY IS NOT NULL
        ) AS SQ
    WHERE RN = 1
),
CLEAN_WAREHOUSE AS (
    SELECT *
    FROM (
        SELECT *
            REPLACE(
                UPPER(TRIM(WAREHOUSE_NAME)) AS WAREHOUSE_NAME,
                UPPER(TRIM(CITY)) AS CITY,
                UPPER(TRIM(PROVINCE)) AS PROVINCE,
                UPPER(TRIM(MANAGER_NAME)) AS MANAGER_NAME
            ),
            ROW_NUMBER() OVER(PARTITION BY WAREHOUSE_KEY ORDER BY WAREHOUSE_KEY) AS RN
        FROM SUPPLY.BRONZE.WAREHOUSE
        WHERE WAREHOUSE_KEY IS NOT NULL
        ) AS SQ
    WHERE RN = 1
),
CLEAN_ORDERS AS (
    SELECT *
    FROM (
        SELECT *
            REPLACE(
                UPPER(TRIM(ORDER_STATUS)) AS ORDER_STATUS
            ),
            ROW_NUMBER() OVER(PARTITION BY ORDER_ID ORDER BY ORDER_ID) AS RN
        FROM SUPPLY.BRONZE.ORDERS
        WHERE ORDER_ID IS NOT NULL
            AND CUSTOMER_KEY IS NOT NULL
            AND PRODUCT_KEY IS NOT NULL
            AND WAREHOUSE_KEY IS NOT NULL
            AND SUPPLIER_KEY IS NOT NULL
        ) AS SQ
    WHERE RN = 1
),
CLEAN_INVENTORY AS (
    SELECT *
FROM (
    SELECT *,
        ROW_NUMBER() OVER(PARTITION BY INVENTORY_ID ORDER BY INVENTORY_ID) AS RN
    FROM SUPPLY.BRONZE.INVENTORY
    WHERE INVENTORY_ID IS NOT NULL
        AND PRODUCT_KEY IS NOT NULL
        AND WAREHOUSE_KEY IS NOT NULL
    ) AS SQ
WHERE RN = 1
),
CLEAN_DATE AS (
    SELECT *
    FROM SUPPLY.BRONZE.DATE
)
SELECT
    O.ORDER_ID, O.ORDER_DATE_KEY, O.CUSTOMER_KEY, O.PRODUCT_KEY, O.WAREHOUSE_KEY, O.SUPPLIER_KEY, QUANTITY_ORDERED, O.UNIT_PRICE, O.DISCOUNT_AMOUNT, O.SHIPPING_COST, ROUND(O.REVENUE, 2) REVENUE,
    
    ROUND(O.QUANTITY_ORDERED * O.UNIT_PRICE, 2) GROSS_AMOUNT,
    ROUND(O.QUANTITY_ORDERED * O.UNIT_PRICE - O.DISCOUNT_AMOUNT - O.SHIPPING_COST, 2) NET_REVENUE,
    ROUND(O.QUANTITY_ORDERED * P.UNIT_COST, 2) TOTAL_COST,
    ROUND((O.QUANTITY_ORDERED * O.UNIT_PRICE - O.DISCOUNT_AMOUNT) - (O.QUANTITY_ORDERED * P.UNIT_COST), 2) GROSS_PROFIT,
    ROUND(((O.QUANTITY_ORDERED * O.UNIT_PRICE - O.DISCOUNT_AMOUNT) - (O.QUANTITY_ORDERED * P.UNIT_COST)) / NULLIF((O.QUANTITY_ORDERED * O.UNIT_PRICE - O.DISCOUNT_AMOUNT),0) * 100, 2) GROSS_PROFIT_MARGIN_PCT,
    ROUND(O.DISCOUNT_AMOUNT / NULLIF(O.QUANTITY_ORDERED * O.UNIT_PRICE, 0) * 100, 2) DISCOUNT_PCT,    
    
    C.CUSTOMER_NAME, C.SEGMENT CUSTOMER_SEGMENT, C.INDUSTRY CUSTOMER_INDUSTRY, C.COUNTRY CUSTOMER_COUNTRY, C.CITY CUSTOMER_CITY, 
    P.PRODUCT_NAME, P.CATEGORY PR_CATEGORY, P.SUBCATEGORY PR_SUBCATEGORY, P.BRAND PR_BRAND,P.UNIT_COST PR_UNIT_COST, P.UNIT_WEIGHT PR_UNIT_WEIGHT, P.SHELF_LIFE_DAYS,
    W.WAREHOUSE_NAME, W.CITY WH_CITY, W.PROVINCE WH_PROVINCE, W.CAPACITY, W.MANAGER_NAME WH_MANAGER_NAME,
    S.SUPPLIER_NAME, S.COUNTRY S_COUNTRY, S.LEAD_TIME_DAYS, S.SUPPLIER_RATING, S.CONTRACT_TYPE,
    D.FULL_DATE, D.DAY, D.MONTH, D.MONTH_NAME, D.QUARTER, D.YEAR, D.WEEK_NUMBER, D.IS_WEEKEND,

    CURRENT_TIMESTAMP::TIMESTAMP_NTZ INGESTION_TIMESTAMP, 'BRONZE' SOURCE_SYSTEM, 'SUPPLY' FILE_NAME
    
FROM CLEAN_ORDERS AS O
LEFT JOIN CLEAN_CUSTOMER AS C ON C.CUSTOMER_KEY = O.CUSTOMER_KEY
LEFT JOIN CLEAN_PRODUCT AS P ON P.PRODUCT_KEY = O.PRODUCT_KEY
LEFT JOIN CLEAN_WAREHOUSE AS W ON W.WAREHOUSE_KEY = O.WAREHOUSE_KEY
LEFT JOIN CLEAN_SUPPLIER AS S ON S.SUPPLIER_KEY = O.SUPPLIER_KEY
LEFT JOIN CLEAN_DATE AS D ON D.DATE_KEY = O.ORDER_DATE_KEY;

SELECT * FROM SUPPLY.SILVER.ORDERS;

-- INVENTORY

CREATE OR REPLACE TABLE SUPPLY.SILVER.INVENTORY AS
WITH CLEAN_PRODUCT AS (
    SELECT *
    FROM (
        SELECT *
            REPLACE(
                UPPER(TRIM(PRODUCT_NAME)) AS PRODUCT_NAME,
                UPPER(TRIM(CATEGORY)) AS CATEGORY,
                UPPER(TRIM(SUBCATEGORY)) AS SUBCATEGORY,
                UPPER(TRIM(BRAND)) AS BRAND
            ),
            ROW_NUMBER() OVER(PARTITION BY PRODUCT_KEY ORDER BY PRODUCT_KEY) AS RN
        FROM SUPPLY.BRONZE.PRODUCT
        WHERE PRODUCT_KEY IS NOT NULL
        ) AS SQ
    WHERE RN = 1
),
CLEAN_WAREHOUSE AS (
    SELECT *
    FROM (
        SELECT *
            REPLACE(
                UPPER(TRIM(WAREHOUSE_NAME)) AS WAREHOUSE_NAME,
                UPPER(TRIM(CITY)) AS CITY,
                UPPER(TRIM(PROVINCE)) AS PROVINCE,
                UPPER(TRIM(MANAGER_NAME)) AS MANAGER_NAME
            ),
            ROW_NUMBER() OVER(PARTITION BY WAREHOUSE_KEY ORDER BY WAREHOUSE_KEY) AS RN
        FROM SUPPLY.BRONZE.WAREHOUSE
        WHERE WAREHOUSE_KEY IS NOT NULL
        ) AS SQ
    WHERE RN = 1
),
CLEAN_INVENTORY AS (
    SELECT *
FROM (
    SELECT *,
        ROW_NUMBER() OVER(PARTITION BY INVENTORY_ID ORDER BY INVENTORY_ID) AS RN
    FROM SUPPLY.BRONZE.INVENTORY
    WHERE INVENTORY_ID IS NOT NULL
        AND PRODUCT_KEY IS NOT NULL
        AND WAREHOUSE_KEY IS NOT NULL
    ) AS SQ
WHERE RN = 1
),
CLEAN_DATE AS (
    SELECT *
    FROM SUPPLY.BRONZE.DATE
)
SELECT
    I.INVENTORY_ID, I.INVENTORY_DATE_KEY, I.PRODUCT_KEY, I.WAREHOUSE_KEY, I.STOCK_ON_HAND, I.REORDER_LEVEL, I.UNITS_ON_ORDER, I.INVENTORY_COST, I.STOCKOUT_FLAG,

    ROUND(I.STOCK_ON_HAND * P.UNIT_COST, 2) STOCK_VALUE,
    ROUND(I.UNITS_ON_ORDER * P.UNIT_COST, 2) ORDER_VALUE,
    I.STOCK_ON_HAND - I.REORDER_LEVEL STOCK_BUFFER,
    ROUND(I.STOCK_ON_HAND / NULLIF(I.REORDER_LEVEL, 0) * 100, 2) STOCK_VS_REORDER_PCT,
    ROUND(I.STOCK_ON_HAND / NULLIF(W.CAPACITY, 0) * 100, 2) WAREHOUSE_UTILIZATION_PCT,
    
    P.PRODUCT_NAME, P.CATEGORY PR_CATEGORY, P.SUBCATEGORY PR_SUBCATEGORY, P.BRAND PR_BRAND,P.UNIT_COST PR_UNIT_COST, P.UNIT_WEIGHT PR_UNIT_WEIGHT, P.SHELF_LIFE_DAYS,
    W.WAREHOUSE_NAME, W.CITY WH_CITY, W.PROVINCE WH_PROVINCE, W.CAPACITY, W.MANAGER_NAME WH_MANAGER_NAME,
    D.FULL_DATE, D.DAY, D.MONTH, D.MONTH_NAME, D.QUARTER, D.YEAR, D.WEEK_NUMBER, D.IS_WEEKEND,

    CURRENT_TIMESTAMP::TIMESTAMP_NTZ INGESTION_TIMESTAMP, 'BRONZE' SOURCE_SYSTEM, 'SUPPLY' FILE_NAME
    
FROM CLEAN_INVENTORY AS I
LEFT JOIN CLEAN_PRODUCT AS P ON P.PRODUCT_KEY = I.PRODUCT_KEY
LEFT JOIN CLEAN_WAREHOUSE AS W ON W.WAREHOUSE_KEY = I.WAREHOUSE_KEY
LEFT JOIN CLEAN_DATE AS D ON D.DATE_KEY = I.INVENTORY_DATE_KEY;
